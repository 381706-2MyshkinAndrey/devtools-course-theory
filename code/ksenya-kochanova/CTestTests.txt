#############################################
##### Testing
#############################################

enable_testing()

add_test( 
    NAME can_Run
    WORKING_DIRECTORY ${bin_dir}
    COMMAND ${appname}
)

add_test( 
    NAME can_Print_Help_Without_Arguments
    WORKING_DIRECTORY ${bin_dir}
    COMMAND ${appname}
)
set_tests_properties (can_Print_Help_Without_Arguments 
  PROPERTIES PASS_REGULAR_EXPRESSION "This is a temperature convertor.")

add_test(
    NAME can_Check_Number_Of_Arguments
    WORKING_DIRECTORY ${bin_dir}
    COMMAND ${appname} 100 0
)
set_tests_properties (can_Check_Number_Of_Arguments 
  PROPERTIES PASS_REGULAR_EXPRESSION "ERROR: Should be 3 arguments: value, oldunit, newunit.")

add_test(
    NAME can_Detect_Wrong_Unit
    WORKING_DIRECTORY ${bin_dir}
    COMMAND ${appname} 100 1 5
)
set_tests_properties (can_Detect_Wrong_Unit_Format 
  PROPERTIES PASS_REGULAR_EXPRESSION "Wrong newunit format!")

add_test(
    NAME can_Detect_Wrong_Value_Format_for_Celsius
    WORKING_DIRECTORY ${bin_dir}
    COMMAND ${appname} -280 0 1
)
set_tests_properties (can_Detect_Wrong_Operation_Format 
  PROPERTIES PASS_REGULAR_EXPRESSION "Wrong value for Celsius!")

add_test(
    NAME can_Detect_Wrong_Value_Format_for_Kelvin
    WORKING_DIRECTORY ${bin_dir}
    COMMAND ${appname} -5 1 2
)
set_tests_properties (can_Detect_Wrong_Operation_Format 
  PROPERTIES PASS_REGULAR_EXPRESSION "Wrong value for Kelvin!")

add_test(
    NAME can_Detect_Wrong_Value_Format_for_Fahrenheit
    WORKING_DIRECTORY ${bin_dir}
    COMMAND ${appname} -500 2 1
)
set_tests_properties (can_Detect_Wrong_Operation_Format 
  PROPERTIES PASS_REGULAR_EXPRESSION "Wrong value for Fahrenheit!")

add_test(
    NAME can_Detect_Wrong_Value_Format_for_Newton
    WORKING_DIRECTORY ${bin_dir}
    COMMAND ${appname} -100 3 1
)
set_tests_properties (can_Detect_Wrong_Operation_Format 
  PROPERTIES PASS_REGULAR_EXPRESSION "Wrong value for Newton!")

add_test(
    NAME can_Convert_Correct_Value
    WORKING_DIRECTORY ${bin_dir}
    COMMAND ${appname} 0 1 0
)
set_tests_properties (can_Add_Positive_Numbers 
  PROPERTIES PASS_REGULAR_EXPRESSION "Result = -273.15")
